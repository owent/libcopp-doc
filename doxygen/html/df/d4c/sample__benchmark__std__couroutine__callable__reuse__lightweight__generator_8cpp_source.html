31using benchmark_callable_future_type = copp::callable_future<int64_t>;
32using benchmark_generator_future_type = copp::generator_lightweight_future<int64_t>;
33
34std::vector<std::unique_ptr<benchmark_callable_future_type>> g_benchmark_callable_list;
35std::vector<benchmark_generator_future_type::context_pointer_type> g_benchmark_generator_list;
36
37int switch_count = 100;
38int max_task_number = 100000;

40benchmark_callable_future_type run_benchmark(size_t idx, int left_switch_count) {
41 int64_t result = 0;
42
43 benchmark_generator_future_type generator{[](benchmark_generator_future_type::context_pointer_type /*ctx*/) {
44 // Do nothing
45 }};
46
47 while (left_switch_count-- >= 0) {
48 generator.get_context()->reset_value();
49
50 g_benchmark_generator_list[idx] = generator.get_context();
51 auto gen_res = co_await generator;
52 result += gen_res;
53 }
54
55 co_return result;
56}

58static void benchmark_round(int index) {
59 g_benchmark_callable_list.reserve(static_cast<size_t>(max_task_number));
60 g_benchmark_generator_list.resize(static_cast<size_t>(max_task_number), nullptr);
61
62 printf("### Round: %d ###\n", index);
63
64 time_t begin_time = time(nullptr);
65 CALC_CLOCK_T begin_clock = CALC_CLOCK_NOW();
66
67 // create coroutines callable
68 while (g_benchmark_callable_list.size() < static_cast<size_t>(max_task_number)) {
69 g_benchmark_callable_list.push_back(std::unique_ptr<benchmark_callable_future_type>(
70 new benchmark_callable_future_type(run_benchmark(g_benchmark_callable_list.size(), switch_count))));
71 }
72
73 time_t end_time = time(nullptr);
74 CALC_CLOCK_T end_clock = CALC_CLOCK_NOW();
75 printf("create %d callable(s) and generator(s), cost time: %d s, clock time: %d ms, avg: %lld ns\n", max_task_number,
76 static_cast<int>(end_time - begin_time), CALC_MS_CLOCK(end_clock - begin_clock),
77 CALC_NS_AVG_CLOCK(end_clock - begin_clock, max_task_number));
78
79 begin_time = end_time;
80 begin_clock = end_clock;
81
82 // yield & resume from runner
83 bool continue_flag = true;
84 long long real_switch_times = static_cast<long long>(0);
85 int32_t round = 0;
86
87 while (continue_flag) {
88 ++round;
89 continue_flag = false;
90 for (auto& generator_context : g_benchmark_generator_list) {
91 benchmark_generator_future_type::context_pointer_type move_context;
92 move_context.swap(generator_context);
93 if (move_context) {
94 move_context->set_value(round);
95 ++real_switch_times;
96 continue_flag = true;
97 }
98 }
99 }
100
101 end_time = time(nullptr);
102 end_clock = CALC_CLOCK_NOW();
103 printf("resume %d callable(s) and generator(s) for %lld times, cost time: %d s, clock time: %d ms, avg: %lld ns\n",
104 max_task_number, real_switch_times, static_cast<int>(end_time - begin_time),
105 CALC_MS_CLOCK(end_clock - begin_clock), CALC_NS_AVG_CLOCK(end_clock - begin_clock, real_switch_times));
106
107 begin_time = end_time;
108 begin_clock = end_clock;
109
110 g_benchmark_callable_list.clear();
111 g_benchmark_generator_list.clear();
112
113 end_time = time(nullptr);
114 end_clock = CALC_CLOCK_NOW();
115 printf("remove %d callable(s), cost time: %d s, clock time: %d ms, avg: %lld ns\n", max_task_number,
116 static_cast<int>(end_time - begin_time), CALC_MS_CLOCK(end_clock - begin_clock),
117 CALC_NS_AVG_CLOCK(end_clock - begin_clock, max_task_number));
118}
