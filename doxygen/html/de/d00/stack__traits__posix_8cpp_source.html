65LIBCOPP_COPP_API std::size_t stack_traits::default_size() LIBCOPP_MACRO_NOEXCEPT {
66 std::size_t size = 8 * minimum_size(); // 64 KB
67 if (is_unbounded()) return size;
68
69 assert(maximum_size() >= minimum_size());
