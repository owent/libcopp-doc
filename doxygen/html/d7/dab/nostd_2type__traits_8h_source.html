203// is_function()
204//
205// Determines whether the passed type `T` is a function type.
206//
207// This metafunction is designed to be a drop-in replacement for the C++11
208// `std::is_function()` metafunction for platforms that have incomplete C++11
209// support (such as libstdc++ 4.x).
210//
211// This metafunction works because appending `const` to a type does nothing to
212// function types and reference types (and forms a const-qualified type
213// otherwise).
214template <class T>
215struct is_function : ::std::integral_constant<bool, !(::std::is_reference<T>::value ||
216 ::std::is_const<typename ::std::add_const<T>::type>::value)> {};
