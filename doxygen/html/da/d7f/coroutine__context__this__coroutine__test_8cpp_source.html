157

159 CASE_EXPECT_EQ(false, jump_to->has_yield);

162 CASE_EXPECT_EQ(true, jump_to->has_yield);
163 CASE_EXPECT_EQ(false, jump_to->has_resume);
164

168 }
169

171 CASE_EXPECT_EQ(ptr, owner);
172
173 CASE_EXPECT_EQ(false, has_yield);
174 CASE_EXPECT_EQ(false, has_resume);
175

177 owner->yield();
178
179 CASE_EXPECT_EQ(true, has_yield);
180 CASE_EXPECT_EQ(true, has_resume);
